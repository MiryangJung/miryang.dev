> Min : *0.01ms, 10.2MB*   
> Max : *3.05ms, 10.3MB*

```python
def solution(answers):
    
    arr1, arr2, arr3 = [1,2,3,4,5], [2,1,2,3,2,4,2,5], [3,3,1,1,2,2,4,4,5,5]
    cnt1, cnt2, cnt3 = 0,0,0
    
    for i in range(len(answers)):
        ans = answers[i]
        cnt1+=ans==arr1[i%5]
        cnt2+=ans==arr2[i%8]
        cnt3+=ans==arr3[i%10]
    
    m = max(cnt1,cnt2,cnt3)
    
    answer = []
    if m == cnt1:
        answer.append(1)
    if m == cnt2:
        answer.append(2)
    if m == cnt3:
        answer.append(3)
    
    return answer
```

> Min : *0.01ms, 10.1MB*   
> Max : *0.01ms, 10.3MB*

```python
def solution(n, lost, reserve):  
    arr = [1]*(n+1)
    
    for n in lost:
        arr[n]=0
    for n in reserve:
        arr[n]+=1
    for i in range(1,len(arr)-1):
        if arr[i]==0 and arr[i-1]==2:
            arr[i]=1
            arr[i-1]=1
        elif arr[i]==0 and arr[i+1]==2:
            arr[i]=1
            arr[i+1]=1
            
    answer = arr.count(1) + arr.count(2) -1
    
    return answer
```

```python
    answer = ''
    for x in s:
        y = ord(x)+n
        if x == " ":
            answer += " "
        elif x.islower() and y > 122:
            answer += chr(y-122+96)
        elif x.isupper() and y > 90:
            answer += chr(y-90+64)
```

```python
        return n
```

## [신규 아이디 추천](https://programmers.co.kr/learn/courses/30/lessons/72410)

> Max : *0.39ms, 10.2MB*

```python
import re

def solution(new_id):
    new_id = new_id.lower()
    new_id = re.sub('[^a-z0-9\-\_\.]','',new_id)
    new_id = re.sub('[\.]+','.',new_id)
    if new_id and new_id[0] == '.':
        new_id=new_id[1:]
    if new_id and new_id[-1] == '.':
        new_id=new_id[:-1]
    if not new_id:
        new_id='a'
    if len(new_id)>=16:
        if new_id[14] == '.':
            new_id=new_id[:14]
        else:
            new_id=new_id[:15]
    elif len(new_id)<=2:
        while len(new_id)!=3:
            new_id+=new_id[-1]
    return new_id
```
